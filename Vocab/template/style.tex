% ============================================
% FIRST PAGE HEADER COMMAND
% ============================================
% Usage: \makefirstpageheader{Author}{Document Title}{Topic Title}
\newcommand{\makefirstpageheader}[3]{%
  % Use TikZ overlay to draw header that extends beyond text margins
  \begin{tikzpicture}[remember picture, overlay]
    % Main red header block - full width, starting from absolute top
    \fill[forumred] 
      (current page.north west) rectangle 
      ([yshift=-2.2cm]current page.north east);
    
    % White "IELTS VOCABULARY" text - baseline close to bottom edge  
    \node[text=white, font=\impactfont\fontsize{48}{58}\selectfont, anchor=base] 
      at ([yshift=-2.05cm]current page.north) 
      {IELTS VOCABULARY};
    
    % Sub-header bar with white border separation
    \fill[white] 
      ([yshift=-2.2cm]current page.north west) rectangle 
      ([yshift=-2.4cm]current page.north east);
    
    \fill[forumred] 
      ([yshift=-2.4cm]current page.north west) rectangle 
      ([yshift=-3.2cm]current page.north east);
    
    % Left text in sub-header - anchor west, position at center
    \node[text=white, font=\sffamily\bfseries\small, anchor=west] 
      at ([xshift=2cm, yshift=-2.8cm]current page.north west) 
      {#1};
    
    % Right text in sub-header - anchor east, position at center
    \node[text=white, font=\sffamily\bfseries\small, anchor=east] 
      at ([xshift=-2cm, yshift=-2.8cm]current page.north east) 
      {#2};
  \end{tikzpicture}%
  
  \vspace{1.5cm} % Space for the header
  
  % Topic title
  \noindent
  {\fontsize{14}{17}\sffamily\bfseries #3}
  
  \vspace{12pt}
}

% ============================================
% VOCABULARY ENTRY FORMATTING
% ============================================
% Usage: \vocabentry{word}{pronunciation}{definition}{example}{translation}
\newcommand{\vocabentry}[5]{%
  \noindent
  {\sffamily\bfseries\color{forumred}\large #1}\\
  {\itshape #2}\\
  #3\\
  {\itshape\textbf{Example:} #4}\\
  {\itshape (#5)}
  
  \vspace{8pt}
}
