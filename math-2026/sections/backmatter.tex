\chapter*{Lộ trình ôn tập gợi ý}
\addcontentsline{toc}{chapter}{Lộ trình ôn tập gợi ý}

\begin{meobotui}{Gợi ý lộ trình 8 tuần}
\begin{enumerate}[leftmargin=2em, itemsep=6pt]
  \item \textbf{Tuần 1--2}: Ôn lại lý thuyết nền + làm câu mức Trung bình.
  \item \textbf{Tuần 3--4}: Luyện câu Khó, tập trung vào chủ đề yếu.
  \item \textbf{Tuần 5--6}: Giải đề tổng hợp, rèn tốc độ.
  \item \textbf{Tuần 7}: Rà soát các câu sai, ôn lại Mẹo nhỏ bỏ túi.
  \item \textbf{Tuần 8}: Thi thử + điều chỉnh tâm lý.
\end{enumerate}
\end{meobotui}
