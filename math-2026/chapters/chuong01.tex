\chapter{Hàm số và ứng dụng}

\section{Tính đơn điệu, cực trị}

% ---- Bài mẫu 1 ----
\begin{baitoan}{1.1 \hfill \tagNguon{Sở GD\&ĐT Hà Nội} \mucDoKho}
Cho hàm số $y = x^3 - 3x^2 + 2$. Tìm tất cả các giá trị thực của tham số $m$ để phương trình $x^3 - 3x^2 + 2 = m$ có ba nghiệm phân biệt.
\end{baitoan}

\begin{loigiai}
Ta xét hàm số $f(x) = x^3 - 3x^2 + 2$.

\textbf{Bước 1:} Tính đạo hàm:
\[ f'(x) = 3x^2 - 6x = 3x(x-2) \]

\textbf{Bước 2:} Bảng biến thiên:

\bbtBacBa{0}{2}{2}{-2}{-\infty}{+\infty}

\textbf{Bước 3:} Đồ thị hàm số $y = f(x)$:

\begin{center}
\begin{tikzpicture}
  \begin{axis}[
    forum graph small,
    xmin=-2, xmax=4,
    ymin=-4, ymax=4,
    xtick={-1,0,1,2,3},
    ytick={-2,0,2},
  ]
    % Đồ thị hàm số
    \addplot[TikzCurve1, domain=-1.5:3.5] {x^3 - 3*x^2 + 2};
    
    % Đường y = m (minh họa m = 1)
    \addplot[TikzCurve2, dashed, domain=-1.5:3.5] {1};
    
    % Điểm cực đại
    \diemDB{0}{2}{$(0;2)$}{above right}
    
    % Điểm cực tiểu
    \diemDB{2}{-2}{$(2;-2)$}{below left}
    
    \legend{$y=x^3-3x^2+2$, $y=m$}
  \end{axis}
\end{tikzpicture}
\end{center}

\textbf{Bước 4:} Kết luận: Phương trình có 3 nghiệm phân biệt khi và chỉ khi:
\[ f(2) < m < f(0) \Leftrightarrow \boxed{-2 < m < 2} \]
\end{loigiai}

\begin{chotdang}{PT $f(x) = m$ có $n$ nghiệm}
Số nghiệm của phương trình $f(x)=m$ bằng số giao điểm của đồ thị $y=f(x)$ và đường thẳng $y=m$. Sử dụng bảng biến thiên để xác định các khoảng giá trị $m$ tương ứng.
\end{chotdang}
