\section{Mô hình dịch bệnh \& Sinh học}

% ===== CÂU 1 =====
\begin{baitoan}{1 \hfill \tagNguon{THPT Thọ Xuân 5 -- Thanh Hóa}}
Sau khi phát hiện bệnh dịch, số người nhiễm bệnh từ ngày đầu đến ngày thứ $t$ là $f(t) = 45t^2 - t^3$ ($t \ge 0$). Đạo hàm $f'(t)$ được xem là tốc độ truyền bệnh (người/ngày).

\begin{enumerate}[label=\alph*), leftmargin=2em]
  \item Đến ngày thứ 45 thì không còn người nhiễm bệnh.
  \item Trong 35 ngày đầu số người nhiễm bệnh luôn tăng.
  \item Tốc độ truyền bệnh tại thời điểm $t$ là $f'(t) = 90t - 3t^2$.
  \item Số người nhiễm bệnh đến ngày thứ 13 là 4752.
\end{enumerate}
\end{baitoan}

\begin{loigiai}
\textbf{a) Đúng.} $f(45) = 45 \cdot 45^2 - 45^3 = 45^3 - 45^3 = 0$.

\textbf{b) Sai.} Số người nhiễm tăng khi $f'(t) > 0$. Ta có:
\[ f'(t) = 90t - 3t^2 = 3t(30 - t). \]

$f'(t) > 0$ khi $0 < t < 30$. Vậy số bệnh nhân tăng trong \textbf{30 ngày đầu}, không phải 35.

\bbtOptim[t]{0}{30}{45}{0}{f_{\max}}{0}{f}{max}

\textbf{c) Đúng.} $f'(t) = (45t^2 - t^3)' = 90t - 3t^2$.

\textbf{d) Sai.} $f(13) = 45 \cdot 13^2 - 13^3 = 45 \cdot 169 - 2197 = 7605 - 2197 = \boxed{5408}$, không phải 4752.
\end{loigiai}

% ===== CÂU 2 =====
\begin{baitoan}{2 \hfill \tagNguon{Sở GD\&ĐT Hưng Yên}}
Chiều cao (cm) của một giống cây trồng tuân theo hàm logistic:
\[ f(t) = \frac{500}{1 + 4e^{-t}}, \]
trong đó $t$ (tháng) tính từ khi nảy mầm. Đạo hàm $f'(t)$ biểu thị tốc độ tăng chiều cao. Hỏi tại thời điểm tốc độ tăng chiều cao lớn nhất, chiều cao cây là bao nhiêu centimet?
\end{baitoan}

\begin{loigiai}
Tốc độ tăng chiều cao:
\[ f'(t) = \frac{500 \cdot 4e^{-t}}{(1 + 4e^{-t})^2} = \frac{2000e^{-t}}{(1 + 4e^{-t})^2}. \]

Để tìm max của $f'(t)$, ta lấy $f''(t) = 0$ (điểm uốn). Đặt $u = e^{-t} > 0$, sau khi tính đạo hàm, điều kiện $f''(t) = 0$ tương đương với
\[ 1 + 4u = 2 \cdot 4u \;\Leftrightarrow\; 4u = 1 \;\Leftrightarrow\; e^{-t} = \frac{1}{4}. \]

Thay vào hàm gốc:
\[ f(t) = \frac{500}{1 + 4 \cdot \frac{1}{4}} = \frac{500}{2} = \boxed{250\;\text{cm}}. \]

Nhận xét: đây là kết quả quen thuộc — hàm logistic $f(t) = \frac{L}{1 + ae^{-kt}}$ luôn có tốc độ tăng lớn nhất tại điểm uốn, nơi $f = \frac{L}{2}$.
\end{loigiai}

\begin{meobotui}{Hàm logistic: max tốc độ tăng tại điểm uốn}
Với hàm logistic $f(t) = \frac{L}{1 + ae^{-kt}}$, tốc độ tăng $f'(t)$ đạt cực đại tại điểm uốn, nơi $f(t) = \frac{L}{2}$.
\end{meobotui}

\section{Mô hình chuyển động \& Vật lý}

% ===== CÂU 3 =====
\begin{baitoan}{3 \hfill \tagNguon{THPT Nguyễn Thị Minh Khai -- Hà Nội}}
Anh Nam làm việc tại giàn khoan cách bờ biển 10\,km. Chị Mai làm tại cơ quan trên bờ cách anh Nam 6\,km phương ngang. Anh Nam đi thuyền (15\,km/h), chị Mai đi bộ (5\,km/h), cùng xuất phát và đến cùng lúc. Hỏi điểm gặp cách cơ quan chị Mai bao nhiêu km? (Làm tròn đến hàng phần trăm).

\end{center}
\end{baitoan}

\begin{loigiai}
Gọi $P$ là điểm gặp trên bờ, cách cơ quan Mai là $x$\,(km). Khi đó chị Mai đi bộ quãng đường $x$, anh Nam đi thuyền quãng đường $NP = \sqrt{10^2 + (6-x)^2}$.

Hai người xuất phát cùng lúc và đến cùng lúc, nên thời gian bằng nhau:
\[ \frac{x}{5} = \frac{\sqrt{100 + (6-x)^2}}{15}. \]

Nhân chéo: $3x = \sqrt{100 + (6-x)^2}$, bình phương hai vế:
\[ 9x^2 = 100 + 36 - 12x + x^2 \Rightarrow 8x^2 + 12x - 136 = 0 \Rightarrow 2x^2 + 3x - 34 = 0. \]

\[ x = \frac{-3 + \sqrt{9 + 272}}{4} = \frac{-3 + \sqrt{281}}{4} \approx \frac{-3 + 16{,}76}{4} \approx \boxed{3{,}44\;\text{km}}. \]
\end{loigiai}

% ===== CÂU 4 =====
\begin{baitoan}{4 \hfill \tagNguon{THPT NK -- LTT -- TP.HCM}}
Hồ nước hình bán nguyệt có đường kính $AB = 150$\,m. Người chèo thuyền từ $A$ đến điểm $C$ trên cung (3\,km/h), nghỉ 2 phút, rồi chạy bộ dọc cung $CB$ đến $B$, sau đó chạy thẳng $BA$ về $A$ (6\,km/h). Hỏi thời gian chậm nhất? (Làm tròn đến hàng phần mười, đơn vị phút).

\begin{center}
\begin{tikzpicture}[scale=0.042, line cap=round, line join=round]
  % Bán nguyệt
  \draw[line width=1pt] (-75,0) arc(180:0:75);
  \draw[line width=1pt] (-75,0) -- (75,0);
  % Điểm A, B
  \fill (-75,0) circle (3pt); \node[below=5pt, font=\small\bfseries] at (-75,0) {$A$};
  \fill (75,0) circle (3pt); \node[below=5pt, font=\small\bfseries] at (75,0) {$B$};
  % Điểm C trên cung
  \coordinate (C) at ({75*cos(150)},{75*sin(150)});
  \fill (C) circle (3pt); \node[above=4pt, xshift=-4pt, font=\small\bfseries] at (C) {$C$};
  % Đường đi
  \draw[blue, line width=0.8pt, ->] (-75,0) -- (C);
  \node[right=6pt, font=\scriptsize, blue] at (-50,28) {thuyền};
  \draw[red, line width=0.8pt, dashed] (C) arc(150:0:75);
  \node[right=4pt, font=\scriptsize, red] at (60,52) {chạy bộ};
  \draw[red, line width=0.8pt, ->] (75,0) -- (-75,0);
  % Góc alpha
  \draw[->] (-75,0) ++(0:25) arc(0:30:25);
  \node[right=2pt, font=\scriptsize] at (-48,10) {$\alpha$};
\end{tikzpicture}
\end{center}
\end{baitoan}

\begin{loigiai}
Đổi đơn vị: $AB = 0{,}15$\,km, bán kính $R = 0{,}075$\,km. Gọi $\alpha = \widehat{CAB}$ (rad).

$AC = AB \cos\alpha = 0{,}15\cos\alpha$; cung $CB = R \cdot 2\alpha = 0{,}15\alpha$; $BA = 0{,}15$.

Thời gian từng chặng:
\begin{itemize}
  \item Chèo thuyền $AC$ (3\,km/h): $t_1 = \frac{0{,}15\cos\alpha}{3} = 0{,}05\cos\alpha$.
  \item Nghỉ: $t_2 = \frac{2}{60} = \frac{1}{30}$ giờ.
  \item Chạy bộ $CB + BA$ (6\,km/h): $t_3 = \frac{0{,}15\alpha + 0{,}15}{6} = 0{,}025(\alpha + 1)$.
\end{itemize}

Tổng thời gian:
\[ T(\alpha) = 0{,}05\cos\alpha + 0{,}025\alpha + 0{,}025 + \frac{1}{30}. \]

Đạo hàm: $T'(\alpha) = -0{,}05\sin\alpha + 0{,}025 = 0 \Rightarrow \sin\alpha = \frac{1}{2} \Rightarrow \alpha = \frac{\pi}{6}$.

Vì $T''(\alpha) = -0{,}05\cos\alpha < 0$, đây là \textit{cực đại} (thời gian chậm nhất):
\[ T\!\left(\frac{\pi}{6}\right) = 0{,}05 \cdot \frac{\sqrt{3}}{2} + 0{,}025 \cdot \frac{\pi}{6} + 0{,}025 + \frac{1}{30} \approx 0{,}1146\;\text{giờ} \approx \boxed{6{,}9\;\text{phút}}. \]
\end{loigiai}

% ===== CÂU 5 =====
\begin{baitoan}{5 \hfill \tagNguon{Cụm trường THPT Hà Tĩnh}}
Chiến sỹ đặc công cần bơi từ bờ $A$ sang điểm $M$ trên đoạn $BC$ ở bờ bên kia, rồi chạy từ $M$ đến $B$. Biết $AC = 60$\,m, $BC = 120$\,m, $AC \perp BC$, vận tốc dòng nước 1\,m/s, vận tốc bơi so với nước 1,5\,m/s, vận tốc chạy 3\,m/s. Thời gian ít nhất? (Làm tròn đến hàng đơn vị, đơn vị giây).

\begin{center}
\begin{tikzpicture}[scale=0.065, line cap=round, line join=round]
  % Hai bờ
  \fill[cyan!6] (-18,-10) rectangle (140,78);
  \draw[cyan!50!black, line width=0.6pt] (-15,0) -- (135,0);

  \fill (0,60) circle (2pt); \node[above left=2pt, font=\small\bfseries] at (0,60) {$A$};
  \fill (120,60) circle (2pt); \node[above right=2pt, font=\small\bfseries] at (120,60) {$C$};
  % Điểm M (trên bờ trên, giữa A và C)
  \fill (50,60) circle (2pt); \node[above=4pt, font=\small\bfseries] at (50,60) {$M$};
  % Đường đi
  \draw[red, line width=0.8pt, ->] (0,0) -- (50,60);
  \node[right=4pt, font=\scriptsize, red] at (30,38) {bơi};
  \draw[blue, line width=0.8pt, ->] (50,60) -- (120,60);
  \node[below=4pt, font=\scriptsize, blue] at (85,60) {chạy};
  % Kích thước dọc
  \draw[<->, line width=0.4pt] (-14,0) -- (-14,60);

  % Vuông góc tại B (góc giữa BA dọc và BC ngang)
  \draw[line width=0.4pt] (7,0) -- (7,7) -- (0,7);

\end{center}

\begin{loigiai}
Vì $AC \perp BC$ và dòng nước chảy dọc bờ, vận tốc thực khi bơi ngang sông:
\[ v_{\text{bơi}} = \sqrt{1{,}5^2 - 1^2} = \sqrt{1{,}25} = \frac{\sqrt{5}}{2}\;\text{m/s}. \]

Đặt $CM = x$ ($0 \le x \le 120$). Quãng đường bơi: $AM = \sqrt{60^2 + x^2} = \sqrt{3600 + x^2}$.

Thời gian tổng cộng:
\[ t(x) = \frac{\sqrt{3600 + x^2}}{\frac{\sqrt{5}}{2}} + \frac{120 - x}{3} = \frac{2\sqrt{3600 + x^2}}{\sqrt{5}} + \frac{120 - x}{3}. \]

Lấy đạo hàm: $t'(x) = \frac{2x}{\sqrt{5}\,\sqrt{3600+x^2}} - \frac{1}{3} = 0$, suy ra $\frac{6x}{\sqrt{5}\,\sqrt{3600+x^2}} = 1$.

Bình phương: $36x^2 = 5(3600 + x^2) \Rightarrow 31x^2 = 18000 \Rightarrow x \approx 24{,}1$.

Thay lại: $t_{\min} \approx \boxed{87\;\text{giây}}$.
\end{loigiai}

% ===== CÂU 6 =====
\begin{baitoan}{6 \hfill \tagNguon{THPT Yên Mô B -- Ninh Bình}}
Một vật chuyển động với phương trình quãng đường $s(t) = -t^3 + 6t^2 + 15t$ (m), $t$ (giây).

\begin{enumerate}[label=\alph*), leftmargin=2em]
  \item Vận tốc tại $t = 2$ là 18\,m/s.
  \item Vận tốc tại thời điểm $t$: $v(t) = -3t^2 + 12t + 15$.
  \item Vật đạt vận tốc lớn nhất tại $t = 2$.
  \item Vật dừng lại sau $t = 4$ giây.
\end{enumerate}
\end{baitoan}

\begin{loigiai}
Vận tốc là đạo hàm của quãng đường: $v(t) = s'(t) = -3t^2 + 12t + 15$.

\textbf{a) Sai.} $v(2) = -3 \cdot 4 + 12 \cdot 2 + 15 = -12 + 24 + 15 = \boxed{27}$\,m/s, không phải 18.

\textbf{b) Đúng.} $v(t) = -3t^2 + 12t + 15$.

\textbf{c) Đúng.} Gia tốc: $a(t) = v'(t) = -6t + 12 = 0 \Rightarrow t = 2$. Vì $v''(2) = -6 < 0$ nên $v$ đạt cực đại tại $t = 2$.

\textbf{d) Sai.} Vật dừng khi $v(t) = 0$:
\[ -3t^2 + 12t + 15 = 0 \Leftrightarrow t^2 - 4t - 5 = 0 \Leftrightarrow (t-5)(t+1) = 0. \]
$t = 5$ (nhận), $t = -1$ (loại). Vật dừng tại $t = 5$, không phải $t = 4$.
\end{loigiai}

\section{Mô hình tài chính}

% ===== CÂU 7 =====
\begin{baitoan}{7 \hfill \tagNguon{THPT Yên Mô B -- Ninh Bình}}
Giá trị còn lại của một chiếc xe: $V(t) = 15000e^{-0{,}15t}$ (USD), $t$ (năm). Hỏi sau bao năm giá trị còn 4500\,USD? (Làm tròn đến hàng đơn vị).
\end{baitoan}

\begin{loigiai}
Ta cần giải $V(t) = 4500$:
\[ 15000e^{-0{,}15t} = 4500 \Leftrightarrow e^{-0{,}15t} = \frac{4500}{15000} = 0{,}3. \]

Lấy logarit tự nhiên: $-0{,}15t = \ln 0{,}3 \approx -1{,}204$.
\[ t = \frac{-\ln 0{,}3}{0{,}15} = \frac{1{,}204}{0{,}15} \approx 8{,}02. \]

Vậy sau khoảng $\boxed{8}$ năm thì xe chỉ còn giá 4500\,USD.
\end{loigiai}

% ===== CÂU 8 =====
\begin{baitoan}{8 \hfill \tagNguon{Cụm liên trường THPT Hải Phòng}}
Năm đầu đi làm, anh Huy lương 10 triệu/tháng. Mỗi năm tăng 12\% so với năm trước. Từ năm thứ 2, mỗi tháng anh cất phần lương tăng so với năm trước để mua ô tô giá 500 triệu (gia đình hỗ trợ 30\%). Hỏi sau ít nhất bao nhiêu năm anh Huy đủ tiền?
\end{baitoan}

\begin{loigiai}
Giá xe anh Huy cần tự lo: $500 - 500 \times 30\% = 350$ triệu.

Lương năm thứ $k$: $L_k = 10 \cdot 1{,}12^{k-1}$ (triệu/tháng).

Từ năm thứ $k$ ($k \ge 2$), phần lương \textit{tăng thêm} mỗi tháng so với năm trước là $L_k - L_{k-1}$. Tiền tiết kiệm cả năm:
\[ S_k = 12(L_k - L_{k-1}) = 12 \cdot 10 \cdot 1{,}12^{k-2}(1{,}12 - 1) = 14{,}4 \cdot 1{,}12^{k-2}. \]

Đây là cấp số nhân với $u_1 = S_2 = 14{,}4$ (triệu), công bội $q = 1{,}12$.

Tổng tiết kiệm từ năm 2 đến năm $n$:
\[ \Sigma_n = 14{,}4 \cdot \frac{1{,}12^{n-1} - 1}{1{,}12 - 1} = 120(1{,}12^{n-1} - 1). \]

Cần $\Sigma_n \ge 350$:
\[ 120(1{,}12^{n-1} - 1) \ge 350 \Leftrightarrow 1{,}12^{n-1} \ge \frac{470}{120} = \frac{47}{12}. \]

Lấy logarit: $n - 1 \ge \log_{1{,}12}\!\frac{47}{12} = \frac{\ln(47/12)}{\ln 1{,}12} \approx \frac{1{,}365}{0{,}1133} \approx 12{,}04$.

Vậy $n \ge 13{,}04$, tức sau ít nhất $\boxed{14}$ năm (kể cả năm đầu không tiết kiệm).
\end{loigiai}

\begin{meobotui}{Dạng tổng cấp số nhân}
Tiền tiết kiệm mỗi năm lập thành CSN $\to$ dùng $S_n = u_1 \cdot \frac{q^n - 1}{q - 1}$, rồi giải BPT logarit.
\end{meobotui}
