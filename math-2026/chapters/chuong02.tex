\section{Chuyển động thẳng trong không gian}

% ===== CÂU 1 =====
\begin{baitoan}{1 \hfill \tagNguon{Nhóm Toán Trong Tâm}}
Trong không gian với hệ trục tọa độ cho trước (đơn vị: km), ra đa phát hiện một chiếc máy bay di chuyển với vận tốc và hướng không đổi từ điểm $A(800;\,500;\,7)$ đến điểm $B(940;\,550;\,9)$ trong 10 phút. Nếu máy bay tiếp tục giữ nguyên vận tốc và hướng bay thì tọa độ của máy bay sau 5 phút tiếp theo là $C(x;\,y;\,z)$. Tính $x + yz$.
\end{baitoan}

\begin{loigiai}
Máy bay bay thẳng đều, nên trong 5 phút nó đi được đúng nửa quãng đường so với 10 phút:
\[ \overrightarrow{BC} = \frac{1}{2}\,\overrightarrow{AB}. \]

Với $\overrightarrow{AB} = (140;\,50;\,2)$, suy ra $\overrightarrow{BC} = (70;\,25;\,1)$, từ đó
\[ C = B + \overrightarrow{BC} = (1010;\,575;\,10). \]

Vậy $x + yz = 1010 + 575 \cdot 10 = \boxed{6760}$.
\end{loigiai}

\textbf{a) Sai.}
$\overrightarrow{SA} = A - S = (30;\,0;\,-30)$, nên
\[ |\overrightarrow{SA}| = \sqrt{30^2 + 0^2 + (-30)^2} = \sqrt{1800} = 30\sqrt{2} \neq 30\sqrt{3}. \]

\textbf{b) Đúng.}
Tính các cạnh đáy: $\overrightarrow{AB} = (-30;\,30;\,0)$ nên $AB = 30\sqrt{2}$.

Tương tự $BC = CD = DA = 30\sqrt{2}$, và $\overrightarrow{AB} \cdot \overrightarrow{AD} = 0$ nên $ABCD$ là hình vuông.

Mặt khác $SA = SB = SC = SD = 30\sqrt{2}$ (các cạnh bên bằng nhau), nên $S.ABCD$ là chóp tứ giác đều.

\textbf{c) Đúng.}
Bốn lực $\vec{F}_1, \ldots, \vec{F}_4$ có độ lớn bằng nhau, đặt từ $S$ đến $A', B', C', D'$ trên các cạnh bên. Do tính đối xứng, $SA' = SB' = SC' = SD'$ và $A'B'C'D'$ vẫn là hình vuông, nên $S.A'B'C'D'$ cũng là chóp tứ giác đều.

\textbf{d) Sai.}
Vì bốn lực bằng nhau và tổng lực bằng trọng lực $\vec{P} = (0;\,0;\,-80)$, suy ra mỗi lực có độ lớn $|\vec{F}_1| = \frac{80}{4} = 20$\,N.

Lực $\vec{F}_1$ hướng từ $S$ đến $A$, tỉ lệ với $\overrightarrow{SA} = (30;\,0;\,-30)$:
\[ \vec{F}_1 = 20 \cdot \frac{(30;\,0;\,-30)}{30\sqrt{2}} = 20 \cdot \frac{(1;\,0;\,-1)}{\sqrt{2}} = (10\sqrt{2};\,0;\,-10\sqrt{2}). \]

Vậy $a + b - c = 10\sqrt{2} + 0 - (-10\sqrt{2}) = 20\sqrt{2} \approx 28{,}3 \neq 20$. Mệnh đề \textbf{sai}.
\end{loigiai}

% ===== CÂU 4 =====
\begin{baitoan}{4 \hfill \tagNguon{Liên trường THPT Hà Nội lần 1}}
Trong không gian $Oxyz$, một vật được treo bằng ba sợi dây không dãn, mỗi sợi có một đầu gắn tại $A(-2;\,-1;\,0)$, $B(2;\,-1;\,0)$, $C(0;\,4;\,0)$, đầu kia gắn với vật tại $D(0;\,0;\,-4)$ (mặt phẳng $(Oxy)$ song song mặt đất, trục $Oz$ hướng lên). Mỗi sợi dây chịu lực căng tối đa 15\,N. Trọng lượng tối đa của vật? (Làm tròn đến hàng phần mười).
\end{baitoan}

\begin{loigiai}
Vật ở $D(0;\,0;\,-4)$ được giữ bởi ba sợi dây kéo về $A$, $B$, $C$ nằm trên mặt phẳng $(Oxy)$, trọng lực kéo xuống theo $\vec{P} = (0;\,0;\,-p)$.

Điều kiện cân bằng: $\vec{T}_A + \vec{T}_B + \vec{T}_C + \vec{P} = \vec{0}$.

Mỗi lực căng hướng từ $D$ đến điểm treo tương ứng:
\begin{align*}
\overrightarrow{DA} &= (-2;\,-1;\,4), \quad DA = \sqrt{4 + 1 + 16} = \sqrt{21}, \\
\overrightarrow{DB} &= (2;\,-1;\,4), \quad DB = \sqrt{21}, \\
\overrightarrow{DC} &= (0;\,4;\,4), \quad DC = 4\sqrt{2}.
\end{align*}

Xét \textbf{thành phần $x$} của phương trình cân bằng:
\[ T_A \cdot \frac{-2}{\sqrt{21}} + T_B \cdot \frac{2}{\sqrt{21}} = 0 \Rightarrow T_A = T_B. \]

Xét \textbf{thành phần $y$}:
\[ T_A \cdot \frac{-1}{\sqrt{21}} + T_B \cdot \frac{-1}{\sqrt{21}} + T_C \cdot \frac{4}{4\sqrt{2}} = 0 \Rightarrow T_C = \frac{2\sqrt{2}}{\sqrt{21}}\,T_A. \]

Vì $\frac{2\sqrt{2}}{\sqrt{21}} < 1$, ta có $T_C < T_A$. Vậy dây $DA$ và $DB$ chịu lực lớn nhất, nên giới hạn ở $T_A = T_B = 15$\,N.

Khi đó $T_C = \dfrac{30\sqrt{2}}{\sqrt{21}}$.

Xét \textbf{thành phần $z$} (trọng lực):
\[ p = T_A \cdot \frac{4}{\sqrt{21}} + T_B \cdot \frac{4}{\sqrt{21}} + T_C \cdot \frac{4}{4\sqrt{2}} = \frac{120}{\sqrt{21}} + \frac{30}{\sqrt{21}} = \frac{150}{\sqrt{21}} \approx \boxed{32{,}7\;\text{N}}. \]
\end{loigiai}

Cùng lúc, kiến đen xuất phát từ $D$ đi thẳng đến $B$ với vận tốc 3\,cm/s. Khoảng cách ngắn nhất của hai chú kiến? (Làm tròn đến hàng phần mười).

\[ \vec{v}_1 = \frac{2}{20\sqrt{2}}\,(0;\,20;\,20) = \left(0;\;\sqrt{2};\;\sqrt{2}\right). \]

Kiến đen: từ $D(0,20,0)$ đến $B(20,0,0)$, $DB = 20\sqrt{2}$, tốc độ 3\,cm/s. Véctơ vận tốc:
\[ \vec{v}_2 = \frac{3}{20\sqrt{2}}\,(20;\,-20;\,0) = \left(\frac{3}{\sqrt{2}};\;-\frac{3}{\sqrt{2}};\;0\right). \]

\begin{align*}
M(t) &= (0;\;\sqrt{2}\,t;\;\sqrt{2}\,t), \\

\[ MN^2(t) = 19t^2 - 100\sqrt{2}\,t + 400. \]

\[ MN_{\min}^2 = 400 - \frac{(100\sqrt{2})^2}{4 \cdot 19} = 400 - \frac{20000}{76} = \frac{2600}{19}. \]

\[ MN_{\min} = \sqrt{\frac{2600}{19}} \approx \boxed{11{,}7\;\text{cm}}. \]
\end{loigiai}

% ===== CÂU 6 =====
\begin{baitoan}{6 \hfill \tagNguon{Cụm trường THPT Hà Tĩnh}}
Trong không gian $Oxyz$ (đơn vị: 10\,km), hai khinh khí cầu $A$ và $B$ bay với $\vec{v}_A = (1;\,2;\,0)$ và $\vec{v}_B = (-2;\,3;\,0)$ (km/h). Tại $t = 0$: $A$ ở $M(5;\,4;\,2)$, $B$ ở $N(6;\,5;\,3)$. Hai khinh khí cầu bay trong 10 giờ.

\begin{enumerate}[label=\alph*), leftmargin=2em]
  \item Sau 3 giờ vị trí khinh khí cầu $A$ là $M'(8;\,10;\,2)$.
  \item Khinh khí cầu $B$ không bay qua $N'(0;\,14;\,3)$.
  \item Khoảng cách hai khinh khí cầu sau 3 giờ là 9\,km.
  \item Khoảng cách ngắn nhất giữa hai khinh khí cầu là 16,1\,km.
\end{enumerate}
\end{baitoan}

\begin{loigiai}
Tại thời điểm $t$ (giờ), vị trí hai khinh khí cầu:
\[ A(t) = (5 + t;\; 4 + 2t;\; 2), \qquad B(t) = (6 - 2t;\; 5 + 3t;\; 3). \]

\textbf{a) Đúng.} Tại $t = 3$: $A(3) = (8;\,10;\,2)$. Đúng như đề cho.

\textbf{b) Sai.} Tại $t = 3$: $B(3) = (6 - 6;\, 5 + 9;\, 3) = (0;\,14;\,3) = N'$. Vậy $B$ \textit{có} bay qua $N'$.

\textbf{c) Sai.} Khoảng cách $M'N'$ trong hệ tọa độ (đơn vị = 10\,km):
\[ M'N' = \sqrt{8^2 + 4^2 + 1^2} = 9 \;\text{(đơn vị)} = 90\;\text{km}, \]
\textit{không phải} 9\,km. Mệnh đề sai.

\textbf{d) Đúng.} Véctơ nối hai khinh khí cầu tại thời điểm $t$:
\[ \overrightarrow{A(t)B(t)} = (1 - 3t;\; 1 + t;\; 1). \]

Khoảng cách (đổi ra km, nhân 10):
\[ d(t) = 10\sqrt{(1 - 3t)^2 + (1+t)^2 + 1} = 10\sqrt{10t^2 - 4t + 3}. \]

Biểu thức trong căn là hàm bậc hai mở lên, đạt min khi $t = \dfrac{4}{20} = 0{,}2$ (giờ):
\[ f(0{,}2) = 10 \cdot 0{,}04 - 4 \cdot 0{,}2 + 3 = 0{,}4 - 0{,}8 + 3 = 2{,}6. \]
\[ d_{\min} = 10\sqrt{2{,}6} \approx \boxed{16{,}1\;\text{km}}. \]
\end{loigiai}

% ===== CÂU 7 =====
\begin{baitoan}{7 \hfill \tagNguon{THPT NK  - LTT -- TP.HCM}}
Một tam giác đều cạnh 1\,m có ba đỉnh $A$, $B$, $C$. Ba chùm sáng LED chạy từ $A$ đến $M$ (3\,cm/s), từ $B$ đến $N$ (2\,cm/s), từ $C$ đến $P$ (1\,cm/s). Gọi $O$ là trọng tâm tam giác. Tính tổng khoảng cách ngắn nhất $OM + ON + OP$ trong một dao động (làm tròn đến hàng phần trăm, đơn vị mét).
\end{baitoan}

\begin{loigiai}
Đổi: $v_M = 0{,}03$\,m/s, $v_N = 0{,}02$\,m/s, $v_P = 0{,}01$\,m/s.

Tại thời điểm $t$ (giây), các điểm sáng cách đỉnh tương ứng: $AM = 0{,}03t$, $BN = 0{,}02t$, $CP = 0{,}01t$.

Chùm sáng chậm nhất (từ $C$, vận tốc 1\,cm/s, cạnh 1\,m = 100\,cm) cần 100\,s để về đích, nên $t \in [0;\,100]$.

Trọng tâm $O$ cách mỗi đỉnh một khoảng $OA = OB = OC = \dfrac{1}{\sqrt{3}} \approx 0{,}577$\,m.

Áp dụng định lý cosin trong tam giác $OAM$ (góc $\widehat{OAM}$ tùy thuộc cách $M$ nằm trên cạnh nào, nhưng do đối xứng tam giác đều, góc này là $30^\circ$):
\begin{align*}
OM^2 &= OA^2 + AM^2 - 2 \cdot OA \cdot AM \cdot \cos 30^\circ \\
     &= \frac{1}{3} + 0{,}0009t^2 - 2 \cdot \frac{1}{\sqrt{3}} \cdot 0{,}03t \cdot \frac{\sqrt{3}}{2} = \frac{1}{3} + 0{,}0009t^2 - 0{,}03t.
\end{align*}

Tương tự: $ON^2 = \frac{1}{3} + 0{,}0004t^2 - 0{,}02t$, \; $OP^2 = \frac{1}{3} + 0{,}0001t^2 - 0{,}01t$.

Tổng $S(t) = OM + ON + OP$ đạt cực tiểu tại khoảng $t \approx 18{,}33$\,s (giải bằng máy tính):
\[ S_{\min} \approx \boxed{1{,}03\;\text{m}}. \]
\end{loigiai}

% ===== CÂU 8 =====
\begin{baitoan}{8 \hfill \tagNguon{Chuyên ĐH KHTN Hà Nội}}
Chọn hệ trục $Oxyz$. Trạm phát sóng tại $A(0;\,1;\,3)$, vùng phủ sóng là hình cầu bán kính 5\,km. Đường thẳng $d:\; \dfrac{x-1}{1} = \dfrac{y}{-1} = \dfrac{z}{2}$.

\begin{enumerate}[label=\alph*), leftmargin=2em]
  \item Véctơ $\vec{u} = (1;\,1;\,2)$ là VTCP của $d$.
  \item Mặt cầu tâm $A$, $R = 5$ có phương trình $x^2 + (y-1)^2 + (z-3)^2 = 25$.
  \item Hình chiếu $H$ của $A$ trên $d$ có hoành độ $\dfrac{2}{3}$.
  \item Đoạn đường trong vùng phủ sóng dài 8,16\,km.
\end{enumerate}
\end{baitoan}

\begin{loigiai}
\textbf{a) Sai.} Từ phương trình $d:\; \frac{x-1}{1} = \frac{y}{-1} = \frac{z}{2}$, VTCP là $\vec{a} = (1;\,-1;\,2)$, \textit{không phải} $(1;\,1;\,2)$.

\textbf{b) Đúng.} Mặt cầu tâm $A(0;\,1;\,3)$, bán kính 5: $x^2 + (y-1)^2 + (z-3)^2 = 25$.

\textbf{c) Sai.} Điểm trên $d$ viết dưới dạng tham số: $H(1+t;\,-t;\,2t)$.

$\overrightarrow{AH} = (1+t;\,-t-1;\,2t-3)$. Điều kiện $\overrightarrow{AH} \perp \vec{a}$:
\begin{align*}
\overrightarrow{AH} \cdot \vec{a} &= (1+t) \cdot 1 + (-t-1)(-1) + (2t-3) \cdot 2 = 0 \\
&= (1+t) + (t+1) + (4t-6) = 6t - 4 = 0 \Rightarrow t = \frac{2}{3}.
\end{align*}

Vậy $x_H = 1 + \frac{2}{3} = \frac{5}{3}$, \textit{không phải} $\frac{2}{3}$.

\textbf{d) Đúng.} Thay tham số $(1+t;\,-t;\,2t)$ vào phương trình mặt cầu:
\[ (1+t)^2 + (-t-1)^2 + (2t-3)^2 = 25. \]
Khai triển:
\[ 6t^2 - 8t + 11 = 25 \Rightarrow 3t^2 - 4t - 7 = 0. \]
\[ t = \frac{4 \pm \sqrt{16 + 84}}{6} = \frac{4 \pm 10}{6} \Rightarrow t_1 = -1, \quad t_2 = \frac{7}{3}. \]

Đoạn đường nằm trong vùng phủ sóng:
\[ BC = |t_2 - t_1| \cdot |\vec{a}| = \frac{10}{3}\,\sqrt{6} \approx \boxed{8{,}16\;\text{km}}. \]
\end{loigiai}
