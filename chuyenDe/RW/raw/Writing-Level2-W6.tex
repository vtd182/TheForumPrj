\cdgreenheading{I. PHÂN TÍCH ĐỀ BÀI VÀ LỰA CHỌN SỐ LIỆU \textit{(ANALYSIS)}}

\cdstep{1. Nhận diện các yếu tố cốt lõi \textit{(General Understanding)}}

Trước khi viết, học viên cần "giải mã" bảng số liệu để tránh nhầm lẫn:
\begin{itemize}
\item \cdkv{Trục thời gian}{3 mốc cụ thể: 1911, 1961 và 2011 (kéo dài 100 năm).}
\item \cdblue{\textbf{Đối tượng so sánh:}}
\begin{itemize}
\item Tổng dân số (Total Population).
\item \cdkv{Nhóm tuổi 15 (Đại diện cho giới trẻ)}{Gồm tỷ lệ \% và số lượng nam/nữ cụ thể.}
\item \cdkv{Nhóm tuổi 75 (Đại diện cho người già)}{Gồm tỷ lệ \% và số lượng nam/nữ cụ thể.}
\end{itemize}
\item \cdblue{\textbf{Đơn vị đo lường:}}
\begin{itemize}
\item \cdkv{Tổng dân số}{Triệu người.}
\item \cdkv{Tỷ lệ}{Phần trăm (\%).}
\item \cdkv{Chi tiết nam/nữ}{Số người (đơn vị hàng trăm nghìn).}
\end{itemize}
\end{itemize}

\cdstep{2. Chọn lọc các điểm nhấn \textit{(Key Features)}}

Để viết Overview và Body tốt, ta cần tìm ra sự thay đổi rõ rệt nhất:
\begin{itemize}
\item \cdkv{Tổng dân số}{Tăng liên tục và mạnh mẽ (từ 36 triệu lên 56 triệu).}
\item \cdblue{\textbf{Xu hướng già hóa (Aging Population):}}
\begin{itemize}
\item Tỷ lệ người trẻ (15 tuổi) giảm dần (1.86\% -> 1.23\%).
\item Tỷ lệ người già (75 tuổi) tăng mạnh (0.23\% -> 0.70\%).
\end{itemize}
\item \cdblue{\textbf{Sự chênh lệch giới tính (Gender Gap):}}
\begin{itemize}
\item \cdkv{Ở tuổi 15}{Số lượng Nam và Nữ khá cân bằng, nhưng về sau Nam vượt nhẹ Nữ.}
\item \cdkv{Ở tuổi 75}{Nữ luôn áp đảo Nam về số lượng (phụ nữ sống thọ hơn).}
\end{itemize}
\end{itemize}

II. CHIẾN LƯỢC TỔ CHỨC VĂN BẢN (STRUCTURE STRATEGY)

Thay vì viết theo từng năm (rất dễ bị liệt kê nhàm chán), ta sẽ nhóm theo độ tuổi. Cách này giúp so sánh sự thay đổi của từng nhóm người qua 100 năm.
\begin{itemize}
\item \cdkv{Body 1}{Tổng dân số và Nhóm 15 tuổi (So sánh xu hướng tăng dân số chung nhưng giảm tỷ lệ người trẻ).}
\item \cdkv{Body 2}{Nhóm 75 tuổi (Tập trung vào sự tăng trưởng mạnh của người già và sự chênh lệch nam/nữ).}
\end{itemize}

III. QUY TRÌNH VIẾT BÀI CỤ THỂ

\cdstep{1. Mở bài \textit{(Introduction)}}
\begin{itemize}
\item \cdkv{Mục đích}{Paraphrase đề bài.}
\item \cdkv{Kỹ thuật nâng cao}{Sử dụng cụm danh từ "demographic statistics" (số liệu nhân khẩu học) và "breakdown" (sự phân chia).}
\end{itemize}

\cdkv{Mẫu}{"The table provides demographic statistics regarding the total population of the UK and the specific breakdown of males and females aged 15 and 75 in three years: 1911, 1961, and 2011."}

\cdstep{2. Tổng quan \textit{(Overview)}}
\begin{itemize}
\item \cdkv{Mục đích}{Nêu 2 xu hướng chính: Dân số tăng và Dân số già đi.}
\item \cdkv{Kỹ thuật nâng cao}{Dùng cấu trúc đối lập While X increased, Y decreased.}
\end{itemize}

\cdkv{Mẫu}{"Overall, the UK witnessed a steady increase in its total population over the century. Additionally, the population became significantly older; the proportion of 75-year-olds rose, whereas the percentage of 15-year-olds showed a downward trend."}

\cdkv{(Dịch}{Nhìn chung, UK chứng kiến sự tăng trưởng dân số đều đặn. Thêm vào đó, dân số trở nên già hơn đáng kể; tỷ lệ người 75 tuổi tăng, trong khi tỷ lệ người 15 tuổi có xu hướng giảm.)}

\cdstep{3. Thân bài 1 (Body Paragraph 1 – Tổng \& Nhóm 15 tuổi)}
\begin{itemize}
\item \cdkv{Hướng triển khai}{Nêu tổng dân số tăng -> Đối lập với tỷ lệ nhóm 15 tuổi giảm \$\textbackslash\{\}to\$ So sánh nam/nữ 15 tuổi.}
\item \cdkv{Phân tích số liệu thông minh}{Đừng chỉ liệt kê. Hãy chỉ ra rằng dù dân số tổng tăng, nhưng tỷ lệ người trẻ lại giảm.}
\end{itemize}

\cdkv{Mẫu}{"In terms of the general population, the figure surged from over 36 million in 1911 to roughly 56 million in 2011. Regarding the younger demographic, although the total number of 15-year-olds remained relatively stable, their percentage relative to the total population dropped consistently, from 1.86\% to 1.23\%. Interestingly, while females slightly outnumbered males in 1911, this trend reversed in later years, with males reaching 354,400 compared to 333,700 females in 2011."}
\begin{itemize}
\item \cdkv{Từ vựng đắt}{Surged (tăng vọt), Dropped consistently (giảm liên tục), Reversed (đảo ngược), Outnumbered (đông hơn).}
\end{itemize}

\cdstep{4. Thân bài 2 (Body Paragraph 2 – Nhóm 75 tuổi)}
\begin{itemize}
\item \cdkv{Hướng triển khai}{Nêu tỷ lệ tăng gấp 3 lần \$\textbackslash\{\}to\$ Số lượng thực tế tăng mạnh \$\textbackslash\{\}to\$ Nhấn mạnh nữ nhiều hơn nam.}
\item \cdkv{Phân tích số liệu thông minh}{Nhấn mạnh sự chênh lệch lớn giữa hai giới ở độ tuổi này.}
\end{itemize}

\cdkv{Mẫu}{"Turning to the elderly age group, the proportion of 75-year-olds experienced a threefold increase, rising from a negligible 0.23\% to 0.70\%. In terms of raw numbers, both genders saw substantial growth, but there was a distinct gender disparity. Females consistently exceeded males in this category. By 2011, the number of 75-year-old women stood at 210,100, which was significantly higher than the male figure of 181,300."}
\begin{itemize}
\item \cdkv{Từ vựng đắt}{Threefold increase (tăng gấp 3), Negligible (không đáng kể - dùng cho số 0.23\%), Gender disparity(chênh lệch giới tính), Exceeded (vượt quá).}
\end{itemize}

\cdstep{2. Cấu trúc câu ăn điểm \textit{(Sentence Structures)}}

\cdblue{\textbf{a. Cấu trúc "While/Whereas" (Mệnh đề đối lập):}}
\begin{itemize}
\item \cdkv{Band 5.0}{The number of old people increased. The number of young people decreased.}
\item \cdkv{Band 6.0+}{The proportion of 75-year-olds rose significantly, whereas the percentage of 15-year-olds declined.}
\end{itemize}

b. Cấu trúc "Participial Phrase" (Rút gọn mệnh đề):
\begin{itemize}
\item \cdkv{Band 5.0}{The figure increased to 56 million. It ended the period at this number.}
\item \cdkv{Band 6.0+}{The figure surged to 56 million, ending the period at the highest point.}
\end{itemize}

\cdblue{\textbf{c. Cấu trúc so sánh gấp lần:}}
\begin{itemize}
\item \cdkv{Band 5.0}{The percentage increased 3 times.}
\item \cdkv{Band 6.0+}{The percentage experienced a threefold increase.}
\end{itemize}
